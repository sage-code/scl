Eve Notepad++ Demo

** example of collection iteration
routine test():
  List this = ["a","b","c","d","e"];
  Integer i = 0;
  Symbol:  e;
process
  cycle: while i <h this.length() loop
    e := this[i];
    i := i + 1;
    case: when e  >= "c" then
        write e;
        if e is not this.tail then
           write ','
        end if;
    end case;
  then
    write ('i = ' + i);
  end cycle;
  print;
return;
